     6	import dev.langchain4j.model.openai.OpenAiChatModel
     7	import kotlinx.coroutines.*
     8	import kotlinx.coroutines.flow.*
     9	import kotlinx.serialization.Serializable

    11	/**
    12	 * Data class representing a message exchanged between agents in the mesh.
    13	 * Inputs and outputs between agents must explicitly and unambiguously describe
    14	 * 'what', 'where', and 'how' the task is intended.
    15	 */
    16	@Serializable
    17	data class MeshMessage(
    18	    val senderId: String,
    19	    val topic: String,
    20	    val content: String,
    21	    val timestamp: Long = System.currentTimeMillis(),
    22	)
    23
    24	interface MeshAgent {
    25	    val id: String
    26	    val roleDescription: String
    27	    suspend fun start(mesh: MeshNetwork)
    28	}
    29
    30	class MeshNetwork(val scope: CoroutineScope) {
    31	    private val _messages = MutableSharedFlow<MeshMessage>(extraBufferCapacity = 100)
    32	    val messages = _messages.asSharedFlow()
    33
    34	    suspend fun broadcast(message: MeshMessage) {
    35	        println("[Network] Broadcast from ${message.senderId} on topic ${message.topic}: ${message.content.take(50)}...")
    36	        _messages.emit(message)
    37	    }
    38	}
    39
    40	abstract class AlphaEvolveAgent(
    41	    override val id: String,
    42	    override val roleDescription: String,
    43	    private val apiKey: String,
    44	) : MeshAgent {
    45	    // using open ai model with a demo key for simulation or real one if provided
    46	    private val model = OpenAiChatModel.builder()
    47	        .apply {
    48	            if (apiKey == "demo") {
    49	                baseUrl("http://langchain4j.dev/demo/openai/v1")
    50	                modelName("gpt-3.5-turbo")
    51	            } else {
    52	                modelName("gpt-4o-mini")
    53	            }
    54	        }
    55	        .apiKey(apiKey)
    56	        .build()
    57
    58	    override suspend fun start(mesh: MeshNetwork) {
    59	        println("[$id] Started and listening to all broadcasts. Role: $roleDescription")
    60	        mesh.messages
    61	            .filter { it.senderId != id }
    62	            .collect { message ->
    63	                // Process each message asynchronously so we don't block the shared flow
    64	                mesh.scope.launch {
    65	                    processMessage(message, mesh)
    66	                }
    67	            }
    68	    }
    69
    70	    private suspend fun processMessage(message: MeshMessage, mesh: MeshNetwork) {
    71	        try {
    72	            // Self-filtering: evaluate if this agent should process the message
    73	            val evalPrompt = """
    74	                You are $id. Your role is: $roleDescription
    75	                A message was broadcast by ${message.senderId} on the topic "${message.topic}".
    76	                Message content: "${message.content}"
    77
    78	                Based on your role, should you actively react to this message?
    79	                Answer strictly with YES or NO.
    80	            """.trimIndent()
    81
    82	            val shouldReactResponse = callModel(evalPrompt).trim()
    83	            if (!shouldReactResponse.equals("YES", ignoreCase = true)) {
    84	                return // Ignored
    85	            }
    86
    87	            println("[$id] Decided to process message from ${message.senderId} on ${message.topic}")
    88
    89	            // AlphaEvolve Algorithm: Reasoning, Critiquing, and Refining
    90	            val context = "Context: Message from ${message.senderId} on ${message.topic}. Content: ${message.content}"
    91
    92	            // Step 1: Reason and generate initial draft
    93	            val draftPrompt = """
    94	                Analyze the following context and propose an initial response or action plan as $id with the role: $roleDescription.
    95	                You are participating in an agent mesh. Your output must explicitly and unambiguously
    96	                describe 'what', 'where', and 'how' the task is intended. Focus on prioritizing
    97	                security, performance, style, documentation, cleanliness, and order. Feel free to ask questions or proactively offer help to clarify ambiguous tasks.
    98
    99	                $context
   100	            """.trimIndent()
   101	            val draft = callModel(draftPrompt)
   102	            println("[$id] AlphaEvolve Draft: ${draft.take(50).replace("\n", " ")}...")
   103
   104	            // Step 2: Critique the draft
   105	            val critiquePrompt = """
   106	                Critique the following draft response to ensure it adheres to security, performance,
   107	                style, documentation, cleanliness, and order. Identify any ambiguities regarding
   108	                'what', 'where', and 'how' the task is intended. The feedback should ensure the mesh continuously evolves and stays up-to-date.
   109
   110	                Context: $context
   111
   112	                Draft:
   113	                $draft
   114	            """.trimIndent()
   115	            val critique = callModel(critiquePrompt)
   116	            println("[$id] AlphaEvolve Critique: ${critique.take(50).replace("\n", " ")}...")
   117
   118	            // Step 3: Refine based on critique
   119	            val refinePrompt = """
   120	                Refine the initial draft based on the critique to produce the final, unambiguous output.
   121	                The final output must explicitly and unambiguously describe 'what', 'where', and 'how'
   122	                the task is intended, and must prioritize security, performance, style, documentation,
   123	                cleanliness, and order.
   124
   125	                IMPORTANT: You must determine the next topic for broadcast to continue the mesh execution.
   126	                End your response exactly with a new line containing ONLY:
   127	                NEXT_TOPIC: <topic_name>
   128
   129	                Draft:
   130	                $draft
   131
   132	                Critique:
   133	                $critique
   134	            """.trimIndent()
   135	            val finalOutput = callModel(refinePrompt)
   136	            println("[$id] AlphaEvolve Final Output Generated.")
   137
   138	            // Determine next topic by parsing the output
   139	            val nextTopicRegex = Regex("""NEXT_TOPIC:\s*(.+)""")
   140	            val matchResult = nextTopicRegex.find(finalOutput)
   141	            val nextTopic = matchResult?.groupValues?.get(1)?.trim()
   142
   143	            if (nextTopic != null) {
   144	                // Strip the NEXT_TOPIC line from the content
   145	                val cleanContent = finalOutput.replace(nextTopicRegex, "").trim()
   146	                mesh.broadcast(
   147	                    MeshMessage(
   148	                        senderId = id,
   149	                        topic = nextTopic,
   150	                        content = cleanContent,
   151	                    ),
   152	                )
   153	            } else {
   154	                println("[$id] Processing complete. No next topic found in output. Broadcasting on 'general' topic as fallback.")
   155	                mesh.broadcast(
   156	                    MeshMessage(
   157	                        senderId = id,
   158	                        topic = "general",
   159	                        content = finalOutput,
   160	                    ),
   161	                )
   162	            }
   163	        } catch (e: Exception) {
   164	            println("[$id] Error processing message: ${e.message}")
   165	            e.printStackTrace()
   166	        }
   167	    }
   168
   169	    private suspend fun callModel(prompt: String): String = withContext(Dispatchers.IO) {
   170	        try {
   171	            if (apiKey == "demo") {
   172	                delay(3000) // Prevent RateLimitException for demo key
   173	            }
   174	            model.chat(prompt)
   175	        } catch (e: Exception) {
   176	            "Simulated response due to model error: ${e.message}. Processing prompt: ${prompt.take(30)}..."
   177	        }
   178	    }
   179	}

   181	class PlannerAgent(apiKey: String) : AlphaEvolveAgent(
   182	    "PlannerAgent",
   183	    "I analyze requests, design plans, and specify tasks unambiguously. If I receive code review feedback, I update the plan.",
   184	    apiKey,
   185	)
   186
   187	class ExecutorAgent(apiKey: String) : AlphaEvolveAgent(
   188	    "ExecutorAgent",
   189	    "I write clean, secure, performant, and well-documented code based on a provided plan. I execute plans and report the code written.",
   190	    apiKey,
   191	)
   192
   193	class CriticAgent(apiKey: String) : AlphaEvolveAgent(
   194	    "CriticAgent",
   195	    "I critically evaluate executed code for security, performance, style, cleanliness, order, and documentation. I provide feedback or approve it.",
   196	    apiKey,
   197	)
   198
   199	fun main() = runBlocking {
   200	    val apiKey = System.getenv("OPENAI_API_KEY") ?: "demo"
   201	    val scope = CoroutineScope(Dispatchers.Default + SupervisorJob())
   202	    val mesh = MeshNetwork(scope)
   203
   204	    val planner = PlannerAgent(apiKey)
   205	    val executor = ExecutorAgent(apiKey)
   206	    val critic = CriticAgent(apiKey)
   207
   208	    println("Starting Agent Mesh Session...")
   209
   210	    // Start agents concurrently
   211	    val plannerJob = scope.launch { planner.start(mesh) }
   212	    val executorJob = scope.launch { executor.start(mesh) }
   213	    val criticJob = scope.launch { critic.start(mesh) }
   214
   215	    // Wait for subscribers to be active
   216	    delay(500)
   217
   218	    // Initiate the mesh with a user request
   219	    mesh.broadcast(
   220	        MeshMessage(
   221	            senderId = "User",
   222	            topic = "user_request",
   223	            content = "Create a new Kotlin class for handling HTTP requests securely and efficiently. Provide clean, well-documented code.",
   224	        ),
   225	    )
   226
   227	    // Let the mesh run for a while
   228	    delay(65000)
   229
   230	    println("Agent Mesh Session Completed.")
   231
   232	    // Cancel child coroutines to prevent hanging
   233	    plannerJob.cancel()
   234	    executorJob.cancel()
   235	    criticJob.cancel()
   236	    scope.cancel()
   237	}
